## Script example

<div className="code-wrap">

```bash
#!/usr/bin/env bash
set -Eeuo pipefail

# Colors
RED=$'\033[31m'
GREEN=$'\033[32m'
YELLOW=$'\033[33m'
RESET=$'\033[0m'

log() { printf "%s[%s]%s %s\n" "$YELLOW" "$(date +%H:%M:%S)" "$RESET" "$*"; }
ok()  { printf "%s✔%s %s\n" "$GREEN" "$RESET" "$*"; }
err() { printf "%s✖%s %s\n" "$RED"   "$RESET" "$*" >&2; }

usage() {
  cat <<'USAGE'
Usage: ./demo.sh [name]
  name   Who to greet (default: world)
USAGE
}

name=${1:-"world"}

log "Starting..."
if [[ -z "${name}" ]]; then
  err "Name must not be empty"
  usage
  exit 1
fi

echo "Hello, ${name}!"
ok "Done"
```
